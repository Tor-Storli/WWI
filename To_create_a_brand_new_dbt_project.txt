# dbt_project.yml
# ---------------

# Name your project! Project names should contain only lowercase characters
# and underscores. A good package name should reflect your organization's
# name or the intended use of these models
name: "wwi_duckdb"
version: "1.0.0"

# This setting configures which "profile" dbt uses for this project.
profile: "wwi_duckdb"

# These configurations specify where dbt should look for different types of files.
# The `model-paths` config, for example, states that models in this project can be
# found in the "models/" directory. You probably won't need to change these!
model-paths: ["wwi_duckdb/models"]
analysis-paths: ["wwi_duckdb/analyses"]
test-paths: ["wwi_duckdb/tests"]
seed-paths: ["wwi_duckdb/seeds"]
macro-paths: ["wwi_duckdb/macros"]
snapshot-paths: ["wwi_duckdb/snapshots"]

clean-targets: # directories to be removed by `dbt clean`
  - "target"
  - "dbt_packages"
  - "logs"

# Configuring models
# Full documentation: https://docs.getdbt.com/docs/configuring-models

# In this example config, we tell dbt to build all models in the example/
# directory as views. These settings can be overridden in the individual model
# files using the `{{ config(...) }}` macro.
models:
  wwi_duckdb:
    +materialized: table
    seeds:
    staging:
      +materialized: view
      +docs:
        node_color: "silver"
    production:
      +materialized: view
      +docs:
        node_color: "silver"
    +docs:
      node_color: "gold"


# Add a schema.yml file to the staging folder:
# --------------------------------------------
version: 1

models:
  - name: stg_city
    columns:
      - name: "City Key"
        tests:
          - unique
          - not_null

  - name: stg_customer
    columns:
      - name: "Customer Key"
        tests:
          - unique
          - not_null

  - name: stg_employee
    columns:
      - name: "Employee Key"
        tests:
          - unique
          - not_null
      - name: "Is Salesperson"  
        tests:
          - accepted_values:
              arguments:
                values: [0, 1]

  - name: stg_paymentmethod
    columns:
      - name: "Payment Method Key"
        tests:
          - unique
          - not_null
  - name: stg_stockitem
    columns:
      - name: "Stock Item Key"
        tests:
          - unique
          - not_null	
  - name: stg_sale
